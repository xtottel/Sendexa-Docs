# Email Verification

Verify email addresses to ensure they're valid, deliverable, and not disposable before sending emails or adding to your database.

---

## Quick Start

```bash
curl -X POST 'https://api.sendexa.co/v1/lookup/email' \
  -H 'Authorization: Bearer sk_live_your_token' \
  -H 'Content-Type: application/json' \
  -d '{
    "email": "user@example.com"
  }'
```

### Response

```json
{
  "success": true,
  "email": "user@example.com",
  "valid": true,
  "deliverable": true,
  "disposable": false,
  "role_based": false,
  "free_provider": false,
  "domain": "example.com",
  "mx_records": true,
  "smtp_check": true,
  "score": 95
}
```

## Code Examples

### JavaScript

```javascript
const axios = require('axios');

async function verifyEmail(email) {
  try {
    const response = await axios.post(
      'https://api.sendexa.co/v1/lookup/email',
      { email },
      {
        headers: {
          Authorization: 'Bearer sk_live_your_token',
          'Content-Type': 'application/json',
        },
      }
    );

    const data = response.data;

    console.log(`Email: ${data.email}`);
    console.log(`Valid: ${data.valid}`);
    console.log(`Deliverable: ${data.deliverable}`);
    console.log(`Score: ${data.score}`);

    return data;
  } catch (error) {
    console.error(
      'Verification failed:',
      error.response?.data || error.message
    );
    throw error;
  }
}

// Single verification
const result = await verifyEmail('user@example.com');

if (result.valid && result.deliverable) {
  console.log('Email is valid and deliverable');
} else {
  console.log('Email is invalid or not deliverable');
}

// Check for disposable
if (result.disposable) {
  console.log('Warning: Disposable email detected');
}

// Check for role-based
if (result.role_based) {
  console.log('Warning: Role-based email (info@, admin@)');
}
```

### Python

```python
import requests

def verify_email(email):
    """Verify an email address"""
    url = 'https://api.sendexa.co/v1/lookup/email'
    headers = {
        'Authorization': 'Bearer sk_live_your_token',
        'Content-Type': 'application/json'
    }
    payload = {'email': email}

    try:
        response = requests.post(url, json=payload, headers=headers)
        response.raise_for_status()
        data = response.json()

        print(f"Email: {data['email']}")
        print(f"Valid: {data['valid']}")
        print(f"Deliverable: {data['deliverable']}")
        print(f"Score: {data['score']}")

        return data
    except requests.exceptions.RequestException as e:
        print(f"Verification failed: {e}")
        raise

# Single verification
result = verify_email('user@example.com')

if result['valid'] and result['deliverable']:
    print('Email is valid')
else:
    print('Email is invalid')

# Check disposable
if result['disposable']:
    print('Warning: Disposable email')

# Check free provider
if result['free_provider']:
    print('Free email provider (Gmail, Yahoo, etc.)')
```

### PHP

```php
<?php

function verifyEmail($email) {
    $url = 'https://api.sendexa.co/v1/lookup/email';
    $token = 'sk_live_your_token';

    $data = ['email' => $email];

    $ch = curl_init($url);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    curl_setopt($ch, CURLOPT_POST, true);
    curl_setopt($ch, CURLOPT_POSTFIELDS, json_encode($data));
    curl_setopt($ch, CURLOPT_HTTPHEADER, [
        'Authorization: Bearer ' . $token,
        'Content-Type: application/json'
    ]);

    $response = curl_exec($ch);
    $httpCode = curl_getinfo($ch, CURLINFO_HTTP_CODE);
    curl_close($ch);

    if ($httpCode === 200) {
        $result = json_decode($response, true);

        echo "Email: " . $result['email'] . "\n";
        echo "Valid: " . ($result['valid'] ? 'Yes' : 'No') . "\n";
        echo "Deliverable: " . ($result['deliverable'] ? 'Yes' : 'No') . "\n";
        echo "Score: " . $result['score'] . "\n";

        return $result;
    } else {
        throw new Exception("Verification failed: " . $response);
    }
}

// Verify email
$result = verifyEmail('user@example.com');

if ($result['valid'] && $result['deliverable']) {
    echo "Email is valid\n";
}

if ($result['disposable']) {
    echo "Warning: Disposable email\n";
}
?>
```

## Common Use Cases

### 1. User Registration

```javascript
app.post('/register', async (req, res) => {
  const { email, name } = req.body;

  // Verify email
  const verification = await verifyEmail(email);

  if (!verification.valid) {
    return res.status(400).json({
      error: 'Invalid email address',
      suggestion: verification.suggestions,
    });
  }

  if (verification.disposable) {
    return res.status(400).json({
      error: 'Disposable email addresses are not allowed',

  }

  if (verification.score < 70) {

  await db.users.create({
    name,
    email,
    email_verified: false,
    email_score: verification.score,
  });

  res.json({ success: true });
});
```

### 3. Email Database Cleaning

```javascript
async function cleanEmailDatabase() {
  const emails = await db.subscribers.findAll();
  const results = {
    valid: [],
    invalid: [],
    disposable: [],
    bouncing: [],
  };

  for (const subscriber of emails) {
    const verification = await verifyEmail(subscriber.email);

    if (!verification.valid || !verification.deliverable) {
      results.invalid.push(subscriber.email);
      await db.subscribers.delete({ where: { id: subscriber.id } });
    } else if (verification.disposable) {
      results.disposable.push(subscriber.email);
      await db.subscribers.delete({ where: { id: subscriber.id } });
    } else if (verification.score < 50) {
      results.bouncing.push(subscriber.email);
      await db.subscribers.update({
        where: { id: subscriber.id },
        data: { status: 'risky', score: verification.score },
      });
    } else {
      results.valid.push(subscriber.email);
      await db.subscribers.update({
        where: { id: subscriber.id },
        data: { status: 'verified', score: verification.score },
      });
    }
  }

  return results;
}
```

### 4. Pre-Send Validation

```javascript
async function sendEmailWithValidation(to, subject, body) {
  // Verify before sending
  const verification = await verifyEmail(to);

  if (!verification.valid || !verification.deliverable) {
    console.log('Skipping invalid email:', to);
    return { sent: false, reason: 'invalid_email' };
  }

  if (verification.score < 60) {
    console.log('Warning: Low score email:', to, verification.score);
  }

  // Send email
  return await sendEmail({
    to,
    subject,
    body,
    tags: {
      email_score: verification.score,
      free_provider: verification.free_provider,
    },
  });
}
```

### 5. Bulk Email Verification

```javascript
async function bulkVerifyEmails(emails) {
  // Split into batches of 100
  const batches = chunkArray(emails, 100);
  const results = [];

  for (const batch of batches) {
    const response = await axios.post(
      'https://api.sendexa.co/v1/lookup/email/bulk',
      { emails: batch },
      {
        headers: {
          Authorization: 'Bearer sk_live_your_token',
          'Content-Type': 'application/json',
        },
      }
    );

    results.push(...response.data.results);
  }

  // Categorize results
  const categorized = {
    valid: results.filter((r) => r.valid && r.deliverable),
    invalid: results.filter((r) => !r.valid),
    disposable: results.filter((r) => r.disposable),
    risky: results.filter((r) => r.score < 50),
  };

  return categorized;
}

// Usage
const emails = await db.subscribers.pluck('email');
const verified = await bulkVerifyEmails(emails);

console.log(`Valid: ${verified.valid.length}`);
console.log(`Invalid: ${verified.invalid.length}`);
console.log(`Disposable: ${verified.disposable.length}`);
console.log(`Risky: ${verified.risky.length}`);
```

### Why Block Disposables?

- **Fraud Prevention** - Temporary emails for abuse
- **Data Quality** - No long-term engagement
- **Compliance** - GDPR issues with temporary data
- **Analytics** - Skews metrics with fake signups

## Catch-All Detection

### What is Catch-All?

A catch-all domain accepts emails to ANY address at that domain, making individual validation impossible.

```javascript
// Example catch-all domain
// Both exist: user@domain.com AND nonexistent@domain.com

if (verification.catch_all) {
  console.log('Catch-all domain detected');
  console.log('Cannot verify individual mailbox');

  // Lower confidence in deliverability
  if (verification.score < 80) {
    console.log('Consider additional verification');
  }
}
```

### 2. Cache Results

```javascript
const cache = new Map();

async function verifyWithCache(email) {
  if (cache.has(email)) {
    return cache.get(email);
  }

  const result = await verifyEmail(email);
  cache.set(email, result);

  return result;
}
```

### 3. Handle Different Risk Levels

```javascript
async function handleEmailByRisk(email) {
  const verification = await verifyEmail(email);

  switch (verification.risk) {
    case 'low':
      // Accept immediately
      return { action: 'accept', verification };

    case 'medium':
      // Accept but flag for monitoring
      return { action: 'accept_flagged', verification };

    case 'high':
      // Require additional verification
      return { action: 'verify_required', verification };
  }
}
```

### 4. Set Score Thresholds

```javascript
const SCORE_THRESHOLDS = {

function categorizeEmail(verification) {
```

## Error Handling

### Common Errors

```javascript
async function safeVerifyEmail(email) {
  try {
    return await verifyEmail(email);
  } catch (error) {
    const code = error.response?.data?.error?.code;

    switch (code) {
      case 'INVALID_EMAIL_FORMAT':
        return { valid: false, error: 'Invalid format' };

      case 'DOMAIN_NOT_FOUND':
        return { valid: false, error: 'Domain does not exist' };

      case 'SMTP_TIMEOUT':
        // Retry once
        return await verifyEmail(email);

      default:
        return { valid: false, error: 'Verification failed' };
    }
  }
}
```
