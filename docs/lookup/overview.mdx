- Network name (MTN, Vodafone, Airtel, etc.)

## Common Use Cases

### 1. Pre-Send Validation

Validate numbers before sending SMS or making calls:

```javascript
// Validate before sending SMS
async function sendSMSWithValidation(phoneNumber, message) {
  // First, validate the number
  const lookup = await validateNumber(phoneNumber);
  
  if (!lookup.valid) {
    console.log('Invalid number, skipping SMS');
    return { error: 'Invalid phone number' };
  }
  
  // Number is valid, proceed with SMS
  return await sendSMS(phoneNumber, message);
}
```

### 2. Database Cleaning

Clean your contact database:

```javascript
// Clean contacts database
async function cleanContactsDatabase() {
  const contacts = await db.contacts.findAll();
  const results = {
    valid: [],
    invalid: [],
    checked: 0
  };
  
  for (const contact of contacts) {
    const lookup = await validateNumber(contact.phone);
    
    if (lookup.valid) {
      results.valid.push(contact);
      // Update with correct format
      await db.contacts.update({
        where: { id: contact.id },
        data: { 
          phone: lookup.international_format,
          carrier: lookup.carrier
        }
      });
    } else {
      results.invalid.push(contact);
      // Mark as invalid
      await db.contacts.update({
        where: { id: contact.id },
        data: { valid: false }
      });
    }
    
    results.checked++;
  }
  
  return results;
}
```

### 3. Carrier-Based Routing

Route messages based on carrier:

```javascript
// Route SMS based on carrier
async function sendSMSWithRouting(phoneNumber, message) {
  const lookup = await carrierLookup(phoneNumber);
  
  // Different routing for different carriers
  const routing = {
    'MTN': 'route_mtn',
    'Vodafone': 'route_vodafone',
    'AirtelTigo': 'route_airteltigo'
  };
  
  const route = routing[lookup.carrier] || 'route_default';
  
  return await sendSMS(phoneNumber, message, { route });
}
```

### 4. User Registration Validation

Validate phone numbers during user registration:

```javascript
// Validate during signup
app.post('/register', async (req, res) => {
  const { phone, email, name } = req.body;
  
  // Validate phone number
  const phoneLookup = await validateNumber(phone);
  
  if (!phoneLookup.valid) {
    return res.status(400).json({
      error: 'Invalid phone number',
      details: phoneLookup.error
    });
  }
  
  // Validate email
  const emailLookup = await verifyEmail(email);
  
  if (!emailLookup.valid) {
    return res.status(400).json({
      error: 'Invalid email address'
    });
  }
  
  // Create user with validated data
  const user = await db.users.create({
    name,
    phone: phoneLookup.international_format,
    email,
    carrier: phoneLookup.carrier,
    line_type: phoneLookup.line_type
  });
  
  res.json({ success: true, user });
});
```

### 5. Bulk Validation

Validate multiple numbers at once:

```javascript
// Bulk validate contacts
async function bulkValidateContacts(phoneNumbers) {
  // Split into batches of 100
  const batches = chunkArray(phoneNumbers, 100);
  const results = [];
  
  for (const batch of batches) {
    const response = await axios.post(
      'https://api.sendexa.co/v1/lookup/bulk',
      { phone_numbers: batch },
      { headers: { Authorization: 'Bearer sk_live_your_token' } }
    );
    
    results.push(...response.data.results);
  }
  
  return results;
}
```

## Best Practices

### 1. Cache Results

```javascript
const cache = new Map();
const CACHE_TTL = 7 * 24 * 60 * 60 * 1000; // 7 days

async function lookupWithCache(phoneNumber) {
  // Check cache
  const cached = cache.get(phoneNumber);
  if (cached && Date.now() - cached.timestamp < CACHE_TTL) {
    return cached.data;
  }
  
  // Lookup
  const result = await carrierLookup(phoneNumber);
  
  // Cache result
  cache.set(phoneNumber, {
    data: result,
    timestamp: Date.now()
  });
  
  return result;
}
```

### 2. Validate Before Sending

Always validate phone numbers before sending messages:

```javascript
// Bad - Send without validation
await sendSMS(phoneNumber, message);

// Good - Validate first
const lookup = await validateNumber(phoneNumber);
if (lookup.valid) {
  await sendSMS(phoneNumber, message);
}
```

### 3. Handle Errors Gracefully

```javascript
async function safeLookup(phoneNumber) {
  try {
    return await carrierLookup(phoneNumber);
  } catch (error) {
    if (error.code === 'RATE_LIMIT_EXCEEDED') {
      // Wait and retry
      await new Promise(r => setTimeout(r, 1000));
      return await carrierLookup(phoneNumber);
    }
    
    // Log error and return default
    console.error('Lookup failed:', error);
    return { valid: false, error: error.message };
  }
}
```

### 4. Use Bulk Lookups

For multiple numbers, use bulk endpoint:

```javascript
// Bad - Individual lookups
for (const number of phoneNumbers) {
  await carrierLookup(number);
}

// Good - Bulk lookup
await bulkLookup(phoneNumbers);
```

### 5. Store Carrier Information

Save carrier info to avoid repeated lookups:

```javascript
// Store carrier info with contact
await db.contacts.create({
  phone: phoneNumber,
  carrier: lookup.carrier,
  line_type: lookup.line_type,
  validated_at: new Date()
});

// Revalidate periodically (e.g., every 90 days)
const contact = await db.contacts.findOne({ phone });
const daysSinceValidation = getDaysSince(contact.validated_at);

if (daysSinceValidation > 90) {
  const lookup = await carrierLookup(contact.phone);
  await db.contacts.update({
    where: { id: contact.id },
    data: { 
      carrier: lookup.carrier,
      validated_at: new Date()
    }
  });
}
```

## Integration Examples

### CRM Integration

```javascript
// Validate when adding contact to CRM
async function addContactToCRM(contactData) {
  const { phone, email, name } = contactData;
  
  // Validate phone
  const phoneLookup = await validateNumber(phone);
  
  if (!phoneLookup.valid) {
    throw new Error('Invalid phone number');
  }
  
  // Validate email
  const emailLookup = await verifyEmail(email);
  
  // Create contact with enriched data
  return await crm.contacts.create({
    name,
    phone: phoneLookup.international_format,
    email,
    carrier: phoneLookup.carrier,
    line_type: phoneLookup.line_type,
    country: phoneLookup.country_name,
    verified: true
  });
}
```

### E-commerce Integration

```javascript
// Validate checkout phone number
app.post('/checkout/validate-phone', async (req, res) => {
  const { phone } = req.body;
  
  const lookup = await validateNumber(phone);
  
  if (!lookup.valid) {
    return res.status(400).json({
      error: 'Please enter a valid phone number',
      suggestion: lookup.suggestion // If available
    });
  }
  
  // Check if mobile (for SMS notifications)
  if (lookup.line_type !== 'mobile') {
    return res.json({
      valid: true,
      warning: 'This appears to be a landline. SMS notifications may not work.',
      phone: lookup.international_format
    });
  }
  
  res.json({
    valid: true,
    phone: lookup.international_format,
    carrier: lookup.carrier
  });
});
```
